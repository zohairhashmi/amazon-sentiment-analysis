#convert summary and reviewText to lower case
instruments_df['reviewText'] = instruments_df['reviewText'].str.lower()

#remove punctuation from summary and reviewText
instruments_df['reviewText'] = instruments_df['reviewText'].str.replace('[^\w\s]','')

#remove numbers from summary and reviewText
instruments_df['reviewText'] = instruments_df['reviewText'].str.replace('\d+', '')

#remove stop words from summary and reviewText
"""
Stop words are basically a set of commonly used words in any language.
They are words that are so common that they are not useful in the context of a search.
"""
stop = stopwords.words('english')
instruments_df['reviewText'] = instruments_df['reviewText'].apply(lambda x: " ".join(x for x in x.split() if x not in stop))

#remove rare words from summary and reviewText
rare = pd.Series(' '.join(instruments_df['reviewText']).split()).value_counts() != 1
rare = rare[rare == False]
instruments_df['reviewText'] = instruments_df['reviewText'].apply(lambda x: " ".join(x for x in x.split() if x not in rare))

#lemmatize summary and reviewText to get the root word
"""
Lemmatization is the process of grouping together the different inflected forms of a word so they can be analysed as a single item.
"""
lemmatizer = WordNetLemmatizer()
instruments_df['reviewText'] = instruments_df['reviewText'].apply(lambda x: " ".join([lemmatizer.lemmatize(word) for word in x.split()]))
